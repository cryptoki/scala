5    /** 
6     * The product from all natural positive numbers, beginning from one to the 
7     * given maximum number. 
8     * 
9     * 4 Factorial or 4! = the product from 1, 2, 3 and 4 = 24. 
10    */ 
11   def factorial(n: Int): Int = { 
12     if (n == 0) 
13       1 
14     else 
15       n * factorial(n - 1) 
16   } 
17   factorial(4)

27    * simple recursive way, but there is a problem. 
28    * the result of every recursion step needs to saved. 
29    * if the depth is to high, a StackOverflowError 
30    * occurs. The program will terminate. 
31    */ 
32   // TODO -me-  write a simple sum example 
33   def overflowingSum(n: Long): Long = { 
34     if (n == 0) 
35       0 
36     else 
37       n + (overflowingSum(n - 1) - (n/2)) 
38   } 
39   overflowingSum(300)
